CREATE TABLE alunos( 
      `matricula`  INT  AUTO_INCREMENT    NOT NULL  , 
      `nome` varchar  (100)   , 
      `cpf` int   , 
      `genero` char   , 
      `graduacao` int   , 
      `Idade` int   , 
      `telefone` int   , 
      `cep` int   , 
      `logradouro` varchar  (50)   , 
      `numero` int   , 
      `bairro` varchar  (50)   , 
      `cidade` varchar  (50)   , 
      `estado` varchar  (2)   , 
 PRIMARY KEY (matricula)) ENGINE=InnoDB DEFAULT CHARSET=utf8 COLLATE=utf8_unicode_ci; 

CREATE TABLE alunoxturmas( 
      `id`  INT  AUTO_INCREMENT    NOT NULL  , 
      `idturmas` int   , 
      `idmatricula` int   , 
 PRIMARY KEY (id)) ENGINE=InnoDB DEFAULT CHARSET=utf8 COLLATE=utf8_unicode_ci; 

CREATE TABLE cursos( 
      `id`  INT  AUTO_INCREMENT    NOT NULL  , 
      `nome` varchar  (50)   , 
 PRIMARY KEY (id)) ENGINE=InnoDB DEFAULT CHARSET=utf8 COLLATE=utf8_unicode_ci; 

CREATE TABLE cursoxdisciplina( 
      `id`  INT  AUTO_INCREMENT    NOT NULL  , 
      `idcursos` int   , 
      `iddisciplinas` int   , 
 PRIMARY KEY (id)) ENGINE=InnoDB DEFAULT CHARSET=utf8 COLLATE=utf8_unicode_ci; 

CREATE TABLE disciplinas( 
      `id`  INT  AUTO_INCREMENT    NOT NULL  , 
      `nome` varchar  (50)   , 
      `faltaspermitidas` int   , 
      `numeroaulas` int   , 
 PRIMARY KEY (id)) ENGINE=InnoDB DEFAULT CHARSET=utf8 COLLATE=utf8_unicode_ci; 

CREATE TABLE graduacao( 
      `id`  INT  AUTO_INCREMENT    NOT NULL  , 
      `nome` varchar  (50)   , 
 PRIMARY KEY (id)) ENGINE=InnoDB DEFAULT CHARSET=utf8 COLLATE=utf8_unicode_ci; 

CREATE TABLE instituicoes( 
      `id`  INT  AUTO_INCREMENT    NOT NULL  , 
      `razaosocial` varchar  (50)   , 
      `cep` int   , 
      `logradouro` varchar  (50)   , 
      `numero` int   , 
      `bairro` varchar  (50)   , 
      `cidade` varchar  (50)   , 
      `estado` varchar  (2)   , 
 PRIMARY KEY (id)) ENGINE=InnoDB DEFAULT CHARSET=utf8 COLLATE=utf8_unicode_ci; 

CREATE TABLE professores( 
      `id`  INT  AUTO_INCREMENT    NOT NULL  , 
      `nome` varchar  (50)   , 
      `cpf` int   , 
      `genero` char   , 
      `graduacao` int   , 
      `idade` int   , 
      `telefone` int   , 
      `cep` int   , 
      `logradouro` varchar  (50)   , 
      `numero` int   , 
      `bairro` varchar  (50)   , 
      `cidade` varchar  (50)   , 
      `estado` varchar  (2)   , 
 PRIMARY KEY (id)) ENGINE=InnoDB DEFAULT CHARSET=utf8 COLLATE=utf8_unicode_ci; 

CREATE TABLE registropresenca( 
      `id`  INT  AUTO_INCREMENT    NOT NULL  , 
      `datapresenca` int   , 
      `presenca` boolean   , 
      `idalunosturma` int   , 
 PRIMARY KEY (id)) ENGINE=InnoDB DEFAULT CHARSET=utf8 COLLATE=utf8_unicode_ci; 

CREATE TABLE turmas( 
      `id`  INT  AUTO_INCREMENT    NOT NULL  , 
      `nome` varchar  (50)   , 
      `idinstituicao` int   , 
      `idcursodisciplina` int   , 
      `idprofessores` int   , 
      `idalunosxturmas` int   , 
 PRIMARY KEY (id)) ENGINE=InnoDB DEFAULT CHARSET=utf8 COLLATE=utf8_unicode_ci;
